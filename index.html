-- ========================================== -- 1. ล้างข้อมูลเก่าและตารางเดิม (เพื่อรีเซ็ตระบบให้สมบูรณ์) -- ========================================== DROP TABLE IF EXISTS public.daily_draws CASCADE; DROP TABLE IF EXISTS public.cards CASCADE; DROP TABLE IF EXISTS public.profiles CASCADE; -- ========================================== -- 2. สร้างโครงสร้างตาราง (Tables) -- ========================================== -- ตารางโปรไฟล์ (เก็บแต้มบุญและรหัสพนักงาน สำหรับหน้าจัดอันดับ) CREATE TABLE public.profiles ( id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY, points INT DEFAULT 0, emp_id TEXT -- เก็บรหัสพนักงานเพื่อนำไปแสดงใน Leaderboard และ Dashboard ); -- ตารางเก็บสำรับไพ่ (ข้อมูลการ์ดทำนายและภารกิจ) CREATE TABLE public.cards ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, "sdgTopic" INT NOT NULL, title TEXT NOT NULL, "fortuneText" TEXT NOT NULL, "missionText" TEXT NOT NULL, "luckyTip" TEXT, "pointsReward" INT DEFAULT 0, "imageUrl" TEXT, "isActive" BOOLEAN DEFAULT true, "createdAt" BIGINT ); -- ตารางประวัติรายวัน (เก็บประวัติการเปิดไพ่, สถานะภารกิจ, รูปหลักฐาน และข้อความเล่าเรื่อง) CREATE TABLE public.daily_draws ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, user_id UUID NOT NULL, "cardId" UUID REFERENCES public.cards(id) ON DELETE CASCADE, "drawnAt" BIGINT, "expiresAt" BIGINT, status TEXT DEFAULT 'pending', "proofImage" TEXT, "missionNote" TEXT -- 🌟 อัปเดตใหม่: เพิ่มช่องเก็บข้อความเล่าเรื่องภารกิจ ); -- ========================================== -- 3. ตั้งค่าความปลอดภัย (Row Level Security - RLS) -- ========================================== -- เปิดระบบความปลอดภัยให้ทุกตาราง ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY; ALTER TABLE public.cards ENABLE ROW LEVEL SECURITY; ALTER TABLE public.daily_draws ENABLE ROW LEVEL SECURITY; -- 🛡️ Policy สำหรับ Cards (ไพ่): -- ให้ทุกคนอ่านและแก้ไขได้ (เพื่อให้เข้าโหมด Admin จากหน้าเว็บและเพิ่มไพ่ได้) CREATE POLICY "Allow public all on cards" ON public.cards FOR ALL USING (true); -- 🛡️ Policy สำหรับ Profiles (ข้อมูลพนักงานและแต้มบุญ): -- 1. อ่านได้ทุกคน: เพื่อให้ระบบดึงแต้มทุกคนมาโชว์ใน "หน้าจัดอันดับ (Leaderboard)" ได้ CREATE POLICY "Allow public read on profiles" ON public.profiles FOR SELECT USING (true); -- 2. เขียน/แก้ไข ได้แค่ตัวเอง: ล็อคให้คนอื่นมาแอบแก้ไขแต้มบุญของเราไม่ได้เด็ดขาด CREATE POLICY "Allow all actions for owner on profiles" ON public.profiles FOR ALL USING (auth.uid() = id); -- 🛡️ Policy สำหรับ Daily Draws (ประวัติการเปิดไพ่และภารกิจ): -- 1. อ่านได้ทุกคน: เพื่อให้ "Admin Dashboard" ดึงสถิติจำนวนคนเปิดไพ่และอ่านข้อความภารกิจมาแสดงได้ CREATE POLICY "Allow public read on daily_draws" ON public.daily_draws FOR SELECT USING (true); -- 2. เขียน/แก้ไข ได้แค่ตัวเอง: ล็อคให้กดยืนยันภารกิจและส่งรูปได้เฉพาะประวัติของรหัสตัวเองเท่านั้น CREATE POLICY "Allow all actions for owner on daily_draws" ON public.daily_draws FOR ALL USING (auth.uid() = user_id);
